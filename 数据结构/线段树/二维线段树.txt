struct node{    int x1,y1,x2,y2;    int ch[4];    int val;}seg[1010*1010*4];int tot;void pushup(int k){    for(int i=0; i<4; i++)    {        int chd = seg[k].ch[i];        if(chd!=-1)            seg[k].val = max(seg[k].val,seg[chd].val);    }}void build(int x1,int y1,int x2,int y2){    tot++;    int k = tot;    seg[k].x1 = x1;seg[k].y1 = y1;    seg[k].x2 = x2;seg[k].y2 = y2;    seg[k].val = -1;    clr(seg[k].ch,-1);    int midx = (x1+x2)>>1;    int midy = (y1+y2)>>1;    if(x1==x2 && y1 == y2)  return;    if(x1<=midx && y1 <= midy)  //左下    {        seg[k].ch[0] = tot+1;        build(x1,y1,midx,midy);    }    if(midx+1 <= x2 && y1 <= midy) //右下    {        seg[k].ch[1] = tot+1;        build(midx+1,y1,x2,midy);    }    if(x1 <= midx && midy+1 <= y2)  //左上    {        seg[k].ch[2] = tot+1;        build(x1,midy+1,midx,y2);    }    if(midx+1 <= x2 && midy+1 <= y2)  //右上    {        seg[k].ch[3] = tot+1;        build(midx+1,midy+1,x2,y2);    }}void modify(int x1,int y1,int x2,int y2,int val,int k){    if(k==-1)   return ;    if(seg[k].x2 < x1 || seg[k].x1 > x2 || seg[k].y2 < y1 ||seg[k].y1 > y2)  return ;    if(seg[k].x1>=x1 && x2 >= seg[k].x2 && seg[k].y1 >=y1 && y2 >= seg[k].y2)    {        seg[k].val = max(val,seg[k].val);        return;    }    for(int i=0; i<4; i++)        modify(x1,y1,x2,y2,val,seg[k].ch[i]);    pushup(k);}int query(int x1,int y1,int x2,int y2,int k){    if(k ==-1)   return -1;    if(seg[k].x2 < x1 || seg[k].x1 > x2 || seg[k].y2 < y1 || seg[k].y1 > y2)  return -1;    if(seg[k].x1>=x1 && x2 >= seg[k].x2 && seg[k].y1 >=y1 && y2 >= seg[k].y2)        return seg[k].val;    int ans = -1;    for(int i=0; i<4; i++)        ans = max(ans,query(x1,y1,x2,y2,seg[k].ch[i]));    return ans;}
